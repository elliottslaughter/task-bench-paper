void compute_kernel(long iterations) {
  double A[64];
  for (int i = 0; i < 64; i++)
    A[i] = 1.2345;
  for (long iter = 0; iter < iterations; iter++)
    for (int i = 0; i < 64; i++)
        A[i] = A[i] * A[i] + A[i];
}
